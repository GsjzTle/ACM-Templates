\textbf{唯一分解定理}

\begin{itemize}
\item
  对一个大于 \(1\) 的整数 \(n\)，\(n\) 可以分解质因数为 \(
  \prod_{i=1}^{k} p_{i}^{a_{i}}=p_{1}^{a_{1}} \cdot p_{2}^{a_{2}} \cdots p_{k}^{a_{k}}
  \)
\end{itemize}

\textbf{约数个数定理}

\begin{itemize}
\item
  \(n\) 的正约数的个数 \(
  d(n)=\prod_{i=1}^{k} (a_{i}+1)=\left(a_{1}+1\right)\left(a_{2}+1\right) \cdots\left(a_{k}+1\right)
  \)
\end{itemize}
